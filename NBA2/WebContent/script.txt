CREATE TABLE  "USERS" 
   (	"USERID" NUMBER(6,0) NOT NULL ENABLE, 
	"USERTYPE" NUMBER(1,0) NOT NULL ENABLE, 
	"USERLEVEL" NUMBER(3,0) NOT NULL ENABLE, 
	"EMAIL" VARCHAR2(4000) NOT NULL ENABLE, 
	"FNAME" VARCHAR2(4000) NOT NULL ENABLE, 
	"MNAME" VARCHAR2(4000), 
	"LNAME" VARCHAR2(4000) NOT NULL ENABLE, 
	"DOB" DATE, 
	"MOBILE" NUMBER(10,0), 
	"ADDRESS" VARCHAR2(4000), 
	"PASSWORD" VARCHAR2(50) NOT NULL ENABLE, 
	 CONSTRAINT "USERS_PK" PRIMARY KEY ("USERID") ENABLE
   )
/
CREATE TABLE  "DEPARTMENTS" 
   (	"DEPTID" NUMBER(2,0) NOT NULL ENABLE, 
	"DEPTFULLNAME" VARCHAR2(200) NOT NULL ENABLE, 
	"DEPTSHORTNAME" VARCHAR2(15) NOT NULL ENABLE, 
	"PRIORITY" NUMBER(2,0), 
	"HODID" NUMBER(6,0) NOT NULL ENABLE, 
	 CONSTRAINT "DEPARTMENTS_PK" PRIMARY KEY ("DEPTID") ENABLE, 
	 CONSTRAINT "DEPARTMENTS_FK" FOREIGN KEY ("HODID")
	  REFERENCES  "USERS" ("USERID") ON DELETE SET NULL ENABLE
   )
/

	"CTID" NUMBER(6,0), 
	 CONSTRAINT "CLASSES_PK" PRIMARY KEY ("CLASSID") ENABLE, 
	 CONSTRAINT "CLASSES_FK" FOREIGN KEY ("DEPTID")
	  REFERENCES  "DEPARTMENTS" ("DEPTID") ON DELETE SET NULL ENABLE, 
	 CONSTRAINT "CLASSES_FK2" FOREIGN KEY ("CRID")
	  REFERENCES  "USERS" ("USERID") ON DELETE SET NULL ENABLE, 
	 CONSTRAINT "CLASSES_FK3" FOREIGN KEY ("CTID")
	  REFERENCES  "USERS" ("USERID") ON DELETE SET NULL ENABLE
   )
/
